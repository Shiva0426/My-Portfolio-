import pandas as pd
from sqlalchemy import create_engine

# Extract data from sources
df = pd.read_csv("flight_delays_raw.csv")

# Clean and transform flights data
df['delay_minutes'] = df['delay_minutes'].fillna(0)
summary = df.groupby('airline').agg({
    'delay_minutes': 'mean',
    'flight_id': 'count'
}).rename(columns={'flight_id': 'total_flights'})

# Load into PostgreSQL warehouse
engine = create_engine('postgresql://admin@db/wh')
summary.to_sql('airline_kpis', engine, if_exists='replace')
print("ETL pipeline executed successfully! ✅")
